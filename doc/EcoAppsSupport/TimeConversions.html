def to_idp_s(format = :default)
  begin
    formatter = I18n.t("time.formats.#{format}", :raise => true) 
    return strftime(formatter).gsub(/0(\d+[月日])/){|t| $1}
  rescue I18n::MissingTranslationData => e
    return to_formatted_s(format)
  end
end
